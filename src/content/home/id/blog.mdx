import BlogPost from "@/components/BlogPost.astro";
import CTA from '@/components/CTA.astro';

import { sortPostsByDate } from "@/scripts/utils.js";
import { getCollection } from "astro:content";
import { getRelativeLocaleUrl } from "astro:i18n";

export const allPosts = await getCollection("blog", 
  (p) => !p.data.is_archived && p.data.lang === frontmatter.lang,
);
export const sortedPosts = sortPostsByDate(allPosts, 3);

<section class="main-section full-columns" id="blog">
  ## Blog
  Tulisan terbaru dari blog saya.

  <ul class="blog-list">
    {
      sortedPosts.map((post) => (
        <BlogPost
          post={post}
          slug={getRelativeLocaleUrl(frontmatter.lang, `blog/${post.slug}`)}
        />
      ))
    }
  </ul>   

  <div class="cta-container">
    <CTA href={getRelativeLocaleUrl(frontmatter.lang, "blog")} margin>Lihat lebih banyak</CTA>
  </div>
</section>
